// URL du fichier généré par ton pipeline GitHub const DATA_URL = "https://raw.githubusercontent.com/kizoka0x/btc-thermal/main/btc_dashboard.json"; // Valeurs par défaut (si GitHub ne répond pas) const DEFAULT = { btcPrice: 0, mayerMultiple: 0, mvrvPct: 0, bullBear30d: 0, bullBear365d: 0, sharpeShort: 0 }; function App() { const [data, setData] = React.useState(DEFAULT); const [status, setStatus] = React.useState("Chargement..."); React.useEffect(() => { fetch(DATA_URL + "?t=" + Date.now()) .then(r => r.json()) .then(d => { setData(d); setStatus("Données live chargées"); console.log("Live:", d); }) .catch(e => { setStatus("Erreur chargement"); console.error(e); }); }, []); return (

BTC Thermal Dashboard

Status: {status}

Prix BTC: ${data.btcPrice}

Mayer: {data.mayerMultiple}

MVRV %: {data.mvrvPct}

BullBear 30d: {data.bullBear30d}

BullBear 365d: {data.bullBear365d}

Sharpe: {data.sharpeShort}

); } ReactDOM.createRoot(document.getElementById("root")).render();
